-- 1. Unified Key Value Store (MUST have to sync everything securely)
CREATE TABLE IF NOT EXISTS sri_finance_store (
    key TEXT PRIMARY KEY,
    value JSONB NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now())
);

-- 2. Customers Table (including Loan Details)
CREATE TABLE IF NOT EXISTS customers (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    "englishName" TEXT,
    phone TEXT,
    address TEXT,
    line TEXT,
    "loanAmount" NUMERIC,
    "interestRate" NUMERIC,
    tenure INTEGER,
    status TEXT DEFAULT 'Active',
    "createdAt" TEXT,
    "fatherName" TEXT,
    occupation TEXT,
    "idProofType" TEXT,
    "idProofNumber" TEXT,
    "referralId" TEXT,
    "referralRelation" TEXT,
    "collectionGroup" TEXT,
    "interestType" TEXT,
    "filesCount" INTEGER,
    "clNo" TEXT,
    "profilePhoto" TEXT
);

-- 3. Collections Table
CREATE TABLE IF NOT EXISTS collections (
    id TEXT PRIMARY KEY,
    "customerId" TEXT,
    "customerName" TEXT,
    amount NUMERIC,
    "interestAmount" NUMERIC,
    "principalAmount" NUMERIC,
    line TEXT,
    date TEXT,
    notes TEXT,
    "paymentMethod" TEXT
);

-- 4. Expenses Table
CREATE TABLE IF NOT EXISTS expenses (
    id TEXT PRIMARY KEY,
    type TEXT,
    amount NUMERIC,
    date TEXT,
    description TEXT
);

-- 5. Employees Table
CREATE TABLE IF NOT EXISTS employees (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    role TEXT,
    phone TEXT,
    status TEXT DEFAULT 'Active',
    "collectionGroup" TEXT,
    salary NUMERIC,
    "joinDate" TEXT,
    attendance JSONB DEFAULT '{}'::jsonb
);

-- 6. Disable Row Level Security (RLS) so clients can safely sync relational ledger data
ALTER TABLE sri_finance_store DISABLE ROW LEVEL SECURITY;
ALTER TABLE customers DISABLE ROW LEVEL SECURITY;
ALTER TABLE collections DISABLE ROW LEVEL SECURITY;
ALTER TABLE expenses DISABLE ROW LEVEL SECURITY;
ALTER TABLE employees DISABLE ROW LEVEL SECURITY;
